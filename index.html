const sources = [ { name: 'HP Series', url: "https://docs.google.com/spreadsheets/d/e/2PACX-1vQ_TkMk68IfYl5ennYorexnetsQ55rU6_yXoHukqFqb1G_Ca57kDLZs1TFLh1e28Uz9QdBRz6bxIfOh/pub?gid=1327187491&single=true&output=csv", color: 'red' }, { name: 'IDC Series', url: "https://docs.google.com/spreadsheets/d/e/2PACX-1vQ_TkMk68IfYl5ennYorexnetsQ55rU6_yXoHukqFqb1G_Ca57kDLZs1TFLh1e28Uz9QdBRz6bxIfOh/pub?gid=1678506384&single=true&output=csv", color: 'blue' } ]; async function performSearch() { const searchWeight = parseFloat(document.getElementById('weightInput').value); const container = document.getElementById('results'); if (isNaN(searchWeight)) { alert("Please enter a numeric weight."); return; } container.innerHTML = "Searching all Indeco lines..."; let resultsHTML = ""; let matchFound = false; for (const source of sources) { try { const response = await fetch(source.url); const csvText = await response.text(); // Cleanup: filter out completely empty rows const rows = csvText.split('\n').map(row => row.split(',')).filter(r => r.length > 1); const headers = rows[0].map(h => h.trim()); for (let colIndex = 1; colIndex < headers.length; colIndex++) { const modelName = headers[colIndex]; if (!modelName || modelName === "") continue; let specs = {}; let minW = 0, maxW = 1000; // Default max high so it shows if range is missing rows.forEach((row) => { const label = row[0]?.trim(); const value = row[colIndex]?.trim(); if (label && value) { specs[label] = value; if (label.toLowerCase().includes("carrier weight")) { // Removes anything that isn't a number, dot, or hyphen const cleanRange = value.replace(/[^0-9.\-]/g, ''); const range = cleanRange.split('-').map(v => parseFloat(v.trim())); minW = range[0]; maxW = range[1] || range[0]; } } }); if (searchWeight >= minW && searchWeight <= maxW) { matchFound = true; resultsHTML += `

${modelName}

Series: ${source.name}

Carrier Weight: ${specs['Carrier weight'] || specs['Carrier Weight'] || 'N/A'}
Operating Weight: ${specs['Operating weight'] || specs['Operating Weight'] || 'N/A'}
`; } } } catch (error) { console.error(`Error fetching ${source.name}:`, error); } } container.innerHTML = matchFound ? resultsHTML : "No equipment matches this weight across all series."; }
